class Main is
    method squareElements(arr: Array[Integer]) is
        var i: Integer
        i := 1
        while i.LessEqual(5) loop
            var x: Integer := arr.get(i)
            arr.set(i, x.Mult(x))
            i := i.Plus(1)
        end
    end
    method main() is
        var numbers: Array
        numbers.set(1, 2)
        numbers.set(2, 3)
        numbers.set(3, 4)
        numbers.set(4, 5)
        numbers.set(5, 6)
        squareElements(numbers)
    end
end
